 //gulpfile
var gulp = require('gulp');
var ts = require('gulp-typescript');
var tsProject = ts.createProject('tsconfig.json');

gulp.task('default', function () {
    return tsProject.src()
        .pipe(tsProject())
        .js.pipe(gulp.dest('dist'));
});

//ts Config
{
  "compilerOptions": {
      "target": "es6",
      "module": "commonjs",
      "outDir": "dist"
  },
  "include": [
      "src/**/*.ts",    
      "bin/**/*.ts"
  ],
  "exclude": [
      "node_modules"
  ]
}
